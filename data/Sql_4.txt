"SELECT DISTINCT "
            + "       BI.ACCOUNT_NO,"
            + "       BI.BILL_REF_NO, "
            + "       BI.BILL_REF_RESETS, "
            + "       BI.PREP_DATE, BI.FROM_DATE, BI.TO_DATE, "
            + "       BI.STATEMENT_DATE, "
            + "       CMF.BILL_COMPANY "
            + "FROM   BILL_INVOICE BI "
            + "       INNER JOIN EXTERNAL_ID_ACCT_MAP EAM ON BI.ACCOUNT_NO = EAM.ACCOUNT_NO "
            + "       INNER JOIN CMF CMF ON CMF.ACCOUNT_NO = BI.ACCOUNT_NO "
            + "WHERE  BI.PREP_ERROR_CODE IS NULL "
            + "AND    BI.FORMAT_ERROR_CODE IS NULL "
            + "AND    BI.ORIG_BILL_REFNO IS NULL "
            + "AND    BI.PREP_STATUS IN (?, ?) "
            + "AND    BI.BACKOUT_STATUS = 0 "
            + "AND    EAM.EXTERNAL_ID = ? "
            + "AND    TRUNC(BI.FROM_DATE) >= ? "
            // From UI, to_date: last day of the bill invoice start date
            // In Kenan, to_date: 1st day of the next month from bill invoice start date, so add -1 to 
            // sync up with UI input
            + "AND    TRUNC(BI.TO_DATE - 1)  <= ? " 
            + "ORDER BY PREP_DATE DESC ";
